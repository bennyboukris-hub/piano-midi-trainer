// --- 1. רישום PWA (Service Worker) --- if ('serviceWorker' in navigator) { window.addEventListener('load', () => { navigator.serviceWorker.register('sw.js') .then(reg => console.log('PWA Service Worker registered!')) .catch(err => console.log('Service Worker registration failed: ', err)); }); } // --- 2. הגדרות בסיסיות של VexFlow --- const { Renderer, Stave, StaveNote, Voice, Formatter, Accidental } = Vex.Flow; let score = 0, highScore = localStorage.getItem('pianoHighScore') || 0; let currentNoteKey = "", canClick = true, countdownInterval, timeLeft = 5, isTimerEnabled = true; document.getElementById('high-score').innerText = highScore; // --- 3. מיפוי תווים ו-MIDI (מפתח פה F2 עד F3) --- const midiMap = { 41: "F/2", 42: "F#/2", 43: "G/2", 44: "G#/2", 45: "A/2", 46: "A#/2", 47: "B/2", 48: "C/3", 49: "C#/3", 50: "D/3", 51: "D#/3", 52: "E/3", 53: "F/3" }; const allNotes = [ { name: "F", oct: 2, type: "white", left: 0 }, { name: "F#", oct: 2, type: "black", left: 8.5 }, { name: "G", oct: 2, type: "white", left: 12.5 }, { name: "G#", oct: 2, type: "black", left: 21 }, { name: "A", oct: 2, type: "white", left: 25 }, { name: "A#", oct: 2, type: "black", left: 33.5 }, { name: "B", oct: 2, type: "white", left: 37.5 }, { name: "C", oct: 3, type: "white", left: 50 }, { name: "C#", oct: 3, type: "black", left: 58.5 }, { name: "D", oct: 3, type: "white", left: 62.5 }, { name: "D#", oct: 3, type: "black", left: 71 }, { name: "E", oct: 3, type: "white", left: 75 }, { name: "F", oct: 3, type: "white", left: 87.5 } ]; // --- 4. פונקציות ליבת המשחק --- function createPiano() { const piano = document.getElementById('piano'); allNotes.forEach(k => { const div = document.createElement('div'); div.className = k.type === "white" ? 'white-key' : 'black-key'; div.style.left = k.left + '%'; div.id = `key-${k.name.replace('#', 's')}-${k.oct}`; div.onclick = () => handleInput(`${k.name}/${k.oct}`, div); piano.appendChild(div); }); } function drawNote() { canClick = true; const div = document.getElementById("staff-output"); div.innerHTML = ""; const renderer = new Renderer(div, Renderer.Backends.SVG); renderer.resize(div.clientWidth, 130); const context = renderer.getContext(); const stave = new Stave(10, 10, div.clientWidth - 20); stave.addClef("bass").setContext(context).draw(); const randomNote = allNotes[Math.floor(Math.random() * allNotes.length)]; currentNoteKey = `${randomNote.name}/${randomNote.oct}`; const note = new StaveNote({ clef: "bass", keys: [currentNoteKey], duration: "q" }); if (randomNote.name.includes("#")) note.addModifier(new Accidental("#")); new Formatter().joinVoices([new Voice({ num_beats: 1, beat_value: 4 }).addTickables([note])]).format([note], 50); note.setContext(context).draw(); if (isTimerEnabled) startCountdown(); } function startCountdown() { timeLeft = 5; document.getElementById('timer-display').innerText = timeLeft; clearInterval(countdownInterval); countdownInterval = setInterval(() => { timeLeft--; document.getElementById('timer-display').innerText = timeLeft; if (timeLeft <= 0) { clearInterval(countdownInterval); if (canClick) resetGame("נגמר הזמן!"); } }, 1000); } function handleInput(guess, el) { if (!canClick || !el) return; if (guess === currentNoteKey) { canClick = false; clearInterval(countdownInterval); el.classList.add('correct'); score += 10; document.getElementById('score').innerText = score; if (score > highScore) { highScore = score; localStorage.setItem('pianoHighScore', highScore); document.getElementById('high-score').innerText = highScore; } setTimeout(() => { el.classList.remove('correct'); drawNote(); }, 500); } else { el.classList.add('wrong'); setTimeout(() => el.classList.remove('wrong'), 300); if (isTimerEnabled) resetGame("טעות!"); else { score = Math.max(0, score - 5); document.getElementById('score').innerText = score; } } } function resetGame(reason) { canClick = false; clearInterval(countdownInterval); score = 0; document.getElementById('score').innerText = score; document.getElementById('msg').innerText = reason + " איפוס!"; document.getElementById('msg').style.color = "#e74c3c"; setTimeout(() => { document.getElementById('msg').innerText = ""; drawNote(); }, 1200); } function toggleTimer() { isTimerEnabled = !isTimerEnabled; const btn = document.getElementById('mode-btn'); const timerText = document.getElementById('timer-display'); clearInterval(countdownInterval); if (isTimerEnabled) { btn.innerText = "מצב: הישרדות (5 שנ')"; btn.className = "toggle-btn"; timerText.classList.remove('timer-off'); drawNote(); } else { btn.innerText = "מצב: אימון (ללא זמן)"; btn.className = "toggle-btn timer-disabled"; timerText.classList.add('timer-off'); } } // --- 5. הגדרות MIDI --- if (navigator.requestMIDIAccess) { navigator.requestMIDIAccess().then(access => { document.getElementById('midi-status').innerText = "MIDI מחובר ומוכן!"; access.inputs.forEach(input => { input.onmidimessage = (m) => { const [cmd, note, vel] = m.data; if (cmd === 144 && vel > 0) { const noteName = midiMap[note]; if (noteName) { const n = allNotes.find(x => `${x.name}/${x.oct}` === noteName); const el = document.getElementById(`key-${n.name.replace('#', 's')}-${n.oct}`); handleInput(noteName, el); } } }; }); }); } // התחלת המשחק createPiano(); drawNote();
